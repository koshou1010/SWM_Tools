# SWM_Tools

python 3.10
## How to start
git clone
cd SWM_Tools
# backend

activate python venv
pip install -r requirements.txt
cd backend
create .env file in ./src
python manage.py runserver
ASGI/Daphne server at http://127.0.0.1:8000/      
# frontend

cd frontend
npm install
npm 
web server at http://localhost:4200/
